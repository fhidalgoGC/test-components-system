import { Switch, Route, useLocation } from "wouter";
import { queryClient } from "./lib/queryClient";
import { QueryClientProvider } from "@tanstack/react-query";
import { Toaster } from "@/components/ui/toaster";
import { TooltipProvider } from "@/components/ui/tooltip";
import {
  AppLanguageProvider,
  LibI18nProvider,
  useAppLanguage,
} from "GC-UI-COMPONENTS";
import { Layout } from "@/components/layout";
import SshKeysPage from "@/pages/ssh-keys";
import LoginPage from "@/pages/login";
import HomePage from "@/pages/home";
import NotFound from "@/pages/not-found";
import { useAuth } from "@/hooks/use-auth";
import { useEffect } from "react";

function AppRouter() {
  const { isAuthenticated, checkAuth } = useAuth();
  const [location] = useLocation();

  useEffect(() => {
    checkAuth();
  }, [location]);

  return (
    <Switch>
      <Route path="/login" component={LoginPage} />
      <Route path="/home" component={HomePage} />
      <Route path="/">
        {isAuthenticated ? (
          <Layout>
            <SshKeysPage />
          </Layout>
        ) : (
          <LoginPage />
        )}
      </Route>
      <Route path="/ssh-keys">
        {isAuthenticated ? (
          <Layout>
            <SshKeysPage />
          </Layout>
        ) : (
          <LoginPage />
        )}
      </Route>
      <Route component={NotFound} />
    </Switch>
  );
}

function AppContent() {
  const appLanguage = useAppLanguage();

  const globalTranslationPaths = [
    { lang: "es", path: "/i18n/es.json" },
    { lang: "en", path: "/i18n/en.json" },
  ];

  return (
    <LibI18nProvider
      parentLanguageProvider={appLanguage}
      globalTranslationPaths={globalTranslationPaths}
      translationPriority="external-first"
    >
      <QueryClientProvider client={queryClient}>
        <TooltipProvider>
          <AppRouter />
          <Toaster />
        </TooltipProvider>
      </QueryClientProvider>
    </LibI18nProvider>
  );
}

export default function App() {
  return (
    <AppLanguageProvider initial="es">
      <AppContent />
    </AppLanguageProvider>
  );
}
